import time
import numpy as np
import pandas as pd
import faiss
from sentence_transformers import SentenceTransformer

from .config import VECTOR_INDEX_PATH, METADATA_PATH, EMBEDDING_MODEL_NAME, TOP_K

class StandardsRetriever:
    def __init__(self):
        self.model = SentenceTransformer(EMBEDDING_MODEL_NAME)
        self.index = faiss.read_index(VECTOR_INDEX_PATH)
        self.metadata = pd.read_csv(METADATA_PATH)
        # Normalize stored vectors if needed
        self.dimension = self.index.d

    def retrieve(self, query: str, top_k: int = TOP_K):
        t0 = time.time()

        query_vec = self.model.encode(query, convert_to_numpy=True)
        query_vec = query_vec.astype("float32")[None, :]
        faiss.normalize_L2(query_vec)

        scores, indices = self.index.search(query_vec, top_k)
        scores = scores[0]
        indices = indices[0]

        results = []
        for score, idx in zip(scores, indices):
            row = self.metadata.iloc[idx]
            results.append({
                "id": row.get("id", idx),
                "title": row.get("title", ""),
                "text": row.get("text", ""),
                "score": float(score),
                "category": row.get("category", "")
            })

        latency = time.time() - t0
        return results, latency
